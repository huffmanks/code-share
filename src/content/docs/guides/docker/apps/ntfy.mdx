  -H "Authorization: Bearer <tk_your_token>" \\
  -H "Title: Test Title" \\
  -H "Tags: rocket,star" \\
  -H "Priority: default" \\
  -d "Message using a token"
`;

export const exampleCli = `
ntfy publish \\
  --host="http://localhost:8383" \\
  --token="<tk_your_token>" \\
  --title="Test Title" \\
  --tags="rocket,star" \\
  --priority="default" \\
  test_topic "Message using a token"
`;
