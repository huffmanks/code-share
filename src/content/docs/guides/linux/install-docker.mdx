import { Code, Steps } from "@astrojs/starlight/components";
export const addDockerSource = `
echo \\
    "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.asc] https://download.docker.com/linux/debian \\
    $(. /etc/os-release && echo "$VERSION_CODENAME") stable" | \\
    sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
`;

<Steps>

1.  <div id="install-dependencies">
      <Code lang="sh" title="Install dependencies" code={`sudo apt install ca-certificates curl -y`} />
    </div>

2.  <div id="install-keyrings">
      <Code lang="sh" title="Install keyrings" code={`sudo install -m 0755 -d /etc/apt/keyrings`} />
    </div>

3.  <div id="add-docker-keyring">
      <Code lang="sh" title="Add Docker keyring" wrap code={`sudo curl -fsSL https://download.docker.com/linux/debian/gpg -o /etc/apt/keyrings/docker.asc`} />
    </div>

4.  <div id="change-mode">
      <Code lang="sh" title="Change mode" code={`sudo chmod a+r /etc/apt/keyrings/docker.asc`} />
    </div>

5.  <div id="add-docker-source-to-apt">
      <Code lang="sh" title="Add docker source to apt" wrap code={addDockerSource} />
    </div>

6.  <div id="apt-update">
      <Code lang="sh" title="Apt update" code={`sudo apt update`} />
    </div>

7.  <div id="install-docker">
      <Code lang="sh" title="Install Docker" wrap code={`sudo apt install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin -y`} />
    </div>
8.  <div id="add-docker-group">
      <Code lang="sh" title="Add docker group" code={`sudo groupadd docker`} />
    </div>
9.  <div id="modify-user">
      <Code lang="sh" title="Modify user" code={`sudo usermod -aG docker $USER`} />
    </div>

10. <div id="reboot">
      <Code lang="sh" title="Reboot" code={`sudo reboot`} />
    </div>

</Steps>
