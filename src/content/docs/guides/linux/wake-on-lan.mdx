<Steps>

1. ## Install ethtool

   <Code lang="sh" code={`sudo apt install ethtool`} />

2. ## Enable WOL

   <Code lang="sh" code={`sudo ethtool -s eth0 wol g`} />

3. ## Send a wake-up packet

   <Code lang="sh" code={`wakeonlan {{SERVER_IP_VAR}}`} />

</Steps>
